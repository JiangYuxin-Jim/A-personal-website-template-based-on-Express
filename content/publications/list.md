# 每条记录有两种写法：
#
# 纯英文论文（无中英文切换）:
#   title  - 论文标题
#   venue  - 期刊/会议
#   doi    - DOI 号（可选）
#   link   - 链接（可选）
#
# 中英双语条目:
#   title_zh / title_en
#   venue_zh / venue_en
#   summary_zh / summary_en  — 用于项目类，替代 venue

  - title_zh: "示例中文论文标题"
    title_en: "Example Paper Title in English"

    venue_en: "Example Journal, 2024"

    title_en: "Projects"

    summary_en: "Describe your projects here, e.g.: Led 2 national projects · Contributed to open source"
